!define TABLE(name,desc) class name as "desc" << (T,#FFAAAA) >>
!define PRIMARY_KEY(x) <b>x</b>
!define FOREIGN_KEY(x) <i>x</i>

TABLE(User, "user") {
  PRIMARY_KEY(id): bigint
  name: string
  email: string
  lastname: string
}

TABLE(Role, "role") {
  PRIMARY_KEY(id): bigint
  name: string
  description: string
  permission_type: string (all | custom)
  permissions: json
}

TABLE(Group, "group") {
  PRIMARY_KEY(id): bigint
  name: string
  description: string
}

TABLE(ContactPerson, "person") {
  PRIMARY_KEY(id): bigint
  FOREIGN_KEY(user_id): bigint
  salutation: string
  first_name: string
  lastname_prefix: string
  last_name: string
  married_name_prefix: string
  married_name: string
  initials: string
  date_of_birth: date
  gender: string
  emails: json
  phones: json
  FOREIGN_KEY(address_id): bigint
  FOREIGN_KEY(created_by): bigint
  FOREIGN_KEY(updated_by): bigint
}

TABLE(Pipeline, "pipeline") {
  PRIMARY_KEY(id): bigint
  name: string
  is_default: boolean
  type: enum // (PipelineType: "lead", "backoffice")
}

TABLE(PipelineStage, "pipeline_stage") {
  PRIMARY_KEY(id): bigint
  FOREIGN_KEY(pipeline_id): bigint
  name: string
  sort_order: int
}

TABLE(Lead, "lead") {
  PRIMARY_KEY(id): bigint
  FOREIGN_KEY(person_id): bigint
  FOREIGN_KEY(user_id): bigint
  FOREIGN_KEY(lead_source_id): bigint
  FOREIGN_KEY(lead_type_id): bigint
  FOREIGN_KEY(lead_pipeline_id): bigint
  FOREIGN_KEY(lead_pipeline_stage_id): bigint
  FOREIGN_KEY(lead_channel_id): bigint
  FOREIGN_KEY(department_id): bigint
  status: string
  lost_reason: text
  expected_close_date: date
  closed_at: datetime
  emails: json
  phones: json
  FOREIGN_KEY(address_id): bigint
  FOREIGN_KEY(created_by): bigint
  FOREIGN_KEY(updated_by): bigint
}

TABLE(Address, "address") {
  PRIMARY_KEY(id): bigint
  FOREIGN_KEY(lead_id): bigint
  FOREIGN_KEY(person_id): bigint
  street: string
  house_number: string
  house_number_suffix: string
  postal_code: string
  state: string
  city: string
  country: string
}

TABLE(Order, "quote (order)") {
  PRIMARY_KEY(id): bigint
  FOREIGN_KEY(lead_id): bigint
  FOREIGN_KEY(pipeline_stage_id): bigint
  FOREIGN_KEY(user_id): bigint
  FOREIGN_KEY(contact_person_id): bigint
  total_amount: decimal
  status: enum
  create_user_id: bigint
}

TABLE(OrderItem, "quote_items") {
    PRIMARY_KEY(id): bigint
    name: string
    quantity: int
    price: float
    discount_percent: float
    discount_amount: int
    tax_percent: float
    tax_amount: int
    total: float
    FOREIGN_KEY(product_id): bigint
    FOREIGN_KEY(quote_id): bigint,
}

TABLE(Product, "product") {
  PRIMARY_KEY(id): bigint
  name: string
  description: string
  price: float
  FOREIGN_KEY(created_by): bigint
  FOREIGN_KEY(updated_by): bigint
}

TABLE(ProductGroup, "product_groups") {
  PRIMARY_KEY(id): bigint
  name: string
  description: string
  FOREIGN_KEY(created_by): bigint
  FOREIGN_KEY(updated_by): bigint
}

TABLE(Backoffice, "backoffice") {
  PRIMARY_KEY(id): bigint
  name: string
  description: text
  FOREIGN_KEY(pipeline_stage_id): bigint
  FOREIGN_KEY(owner_user_id): bigint
}

TABLE(Activity, "activity") {
  PRIMARY_KEY(id): bigint
  title: string
  type: string
  location: string
  comment: text
  additional: json
  schedule_from: datetime
  schedule_to: datetime
  is_done: boolean
  FOREIGN_KEY(user_id): bigint
  FOREIGN_KEY(group_id): bigint
}

TABLE(Anamnesis, "anamnesis") {
  PRIMARY_KEY(id): bigint
  title: string
  type: string
  location: string
  comment: text
  additional: json
  schedule_from: datetime
  schedule_to: datetime
  is_done: boolean
  FOREIGN_KEY(user_id): bigint
  FOREIGN_KEY(lead_id): bigint
}

note right of Pipeline
  Sales funnel with ordered stages for lead.
  Initial records:
  - type lead; Hernia, Private scan
  - type backoffice; Hernia, Private scan
  Each pipeline has it's own stages.
end note

note right of Lead
  Potential client optionally linked to a user and contact person.
  Pipeline stages ("new", "in behandeling", "afgehandeld").
end note

note right of Order
  Closed deal, linked to a lead, user, and contact person.
end note

note right of Backoffice
  Automation flow attached to a pipeline stage.
end note

note right of Activity
  Call | Task or automated step in a Backoffice, optionally assigned to a user.
end note

note right of ContactPerson
  External contact involved in leads or orders, not a user in the system.
end note

<script>
document.addEventListener("DOMContentLoaded", () => {
  setTimeout(() => {
    const menuContainer = document.querySelector(".dynamic-menu");
    const headers = [...document.querySelectorAll("h2, h3")]; // Let op: aangepast naar h2 en h3

    if (!menuContainer || headers.length === 0) {
      console.warn("Menu container or headers not found.");
      return;
    }

    let currentDropdown = null;

    headers.forEach(header => {
      if (header.tagName === "H2") {
        console.info('-', header.textContent);
    } else {
      console.info('--', header.textContent);
    }
      if (!header.id) log.error('header has no id', header);

      if (header.tagName === "H2") {
        // nieuwe hoofdsectie
        currentDropdown = document.createElement("div");
        currentDropdown.className = "dropdown";

        const mainLink = document.createElement("a");
        mainLink.href = `#${header.id}`;
        mainLink.textContent = `${header.textContent} ▾`;

        const dropdownContent = document.createElement("div");
        dropdownContent.className = "dropdown-content";

        currentDropdown.appendChild(mainLink);
        currentDropdown.appendChild(dropdownContent);
        menuContainer.appendChild(currentDropdown);
      }

      if (header.tagName === "H3" && currentDropdown) {
        const subLink = document.createElement("a");
        subLink.href = `#${header.id}`;
        subLink.textContent = header.textContent;
        currentDropdown.querySelector(".dropdown-content").appendChild(subLink);
      }
    });
  }, 100);
});
</script>

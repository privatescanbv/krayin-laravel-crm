== Database Ontwerp

[plantuml]
....
@startuml

!define table(x) class x << (T,#FFAAAA) >>
!define primary_key(x) <u>x</u>
!define foreign_key(x) #x#

skinparam class {
    BackgroundColor White
    ArrowColor Black
    BorderColor Black
}

table(Employee) {
    primary_key(id): bigint
    first_name: string
    last_name: string
    email: string
    phone: string
    department: string
    position: string
    status: enum
    notes: text
    created_at: timestamp
    updated_at: timestamp
}

table(Task) {
    primary_key(id): bigint
    title: string
    description: text
    foreign_key(assigned_to): bigint
    foreign_key(created_by): bigint
    priority: enum
    status: enum
    due_date: date
    completed_at: timestamp
    notes: text
    created_at: timestamp
    updated_at: timestamp
}

table(Lead) {
    primary_key(id): bigint
    company_name: string
    foreign_key(contact_person_id): bigint
    email: string
    phone: string
    address: string
    city: string
    postal_code: string
    country: string
    status: enum
    source: string
    notes: text
    created_at: timestamp
    updated_at: timestamp
}

table(ContactPerson) {
    primary_key(id): bigint
    first_name: string
    last_name: string
    email: string
    phone: string
    mobile: string
    position: string
    department: string
    is_primary: boolean
    notes: text
    created_at: timestamp
    updated_at: timestamp
}

table(Offer) {
    primary_key(id): bigint
    foreign_key(lead_id): bigint
    offer_number: string
    valid_until: date
    status: enum
    total_amount: decimal
    vat_amount: decimal
    notes: text
    created_at: timestamp
    updated_at: timestamp
}

table(Order) {
    primary_key(id): bigint
    foreign_key(offer_id): bigint
    order_number: string
    order_date: date
    delivery_date: date
    status: enum
    total_amount: decimal
    vat_amount: decimal
    shipping_address: text
    billing_address: text
    payment_status: enum
    notes: text
    created_at: timestamp
    updated_at: timestamp
}

table(Product) {
    primary_key(id): bigint
    foreign_key(supplier_id): bigint
    sku: string
    name: string
    description: text
    price: decimal
    vat_rate: decimal
    stock: integer
    min_stock: integer
    category: string
    status: enum
    is_resource: boolean
    capacity_per_day: integer
    duration_minutes: integer
    created_at: timestamp
    updated_at: timestamp
}

table(ResourceBooking) {
    primary_key(id): bigint
    foreign_key(product_id): bigint
    foreign_key(order_item_id): bigint
    start_time: timestamp
    end_time: timestamp
    status: enum
    notes: text
    created_at: timestamp
    updated_at: timestamp
}

table(Supplier) {
    primary_key(id): bigint
    company_name: string
    contact_person: string
    email: string
    phone: string
    address: string
    city: string
    postal_code: string
    country: string
    vat_number: string
    payment_terms: string
    status: enum
    notes: text
    created_at: timestamp
    updated_at: timestamp
}

table(OrderItem) {
    primary_key(id): bigint
    foreign_key(order_id): bigint
    foreign_key(product_id): bigint
    quantity: integer
    unit_price: decimal
    vat_rate: decimal
    total_price: decimal
    scheduled_start: timestamp
    scheduled_end: timestamp
    created_at: timestamp
    updated_at: timestamp
}

Lead "1" -- "0..*" Offer
Offer "1" -- "0..1" Order
Order "1" -- "0..*" OrderItem
OrderItem "0..*" -- "1" Product
Supplier "1" -- "0..*" Product
Lead "1" -- "0..*" ContactPerson
Employee "1" -- "0..*" Task
Task "0..*" -- "1" Employee
Product "1" -- "0..*" ResourceBooking
OrderItem "1" -- "0..*" ResourceBooking

=== Relaties
[cols="1,2"]
|===
|Relatie |Beschrijving
|Lead -> Offer |Een lead kan meerdere offertes hebben
|Offer -> Order |Een offerte kan één order hebben
|Order -> Product |Een order kan meerdere producten bevatten (via order_items)
|Supplier -> Product |Een leverancier kan meerdere producten hebben
